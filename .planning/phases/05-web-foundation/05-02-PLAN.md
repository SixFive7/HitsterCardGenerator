  }

<task type="auto">
  <name>Task 2: Add Tailwind CSS v4</name>
  <files>web/package.json, web/vite.config.ts, web/src/app.css</files>
  <action>
Install Tailwind v4 with Vite plugin:
```bash
cd web && npm install -D tailwindcss @tailwindcss/vite
```

Update vite.config.ts to add tailwindcss plugin:
```typescript
import tailwindcss from '@tailwindcss/vite'

export default defineConfig({
  plugins: [svelte(), tailwindcss()],
  // ... rest of config
})
```

Create/update src/app.css with Tailwind v4 syntax:
```css
@import "tailwindcss";

/* Custom theme for Hitster - fun, playful, bright */
@theme {
  --color-primary: #1DB954;      /* Spotify green */
  --color-primary-dark: #169c46;
  --color-accent: #FF6B6B;       /* Playful coral */
  --color-accent-dark: #ee5a5a;
  --color-surface: #191414;      /* Spotify dark */
  --color-surface-light: #282828;
}
```

Ensure app.css is imported in main.ts.
  </action>
  <verify>npm run dev shows no Tailwind errors, Tailwind classes work in App.svelte</verify>
  <done>Tailwind v4 working with custom theme colors</done>
</task>
